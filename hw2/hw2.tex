\begin{answer}{1}

\paragraph{a)} Every continent that is not Australia or Antarctica is connected to another continent. 

\begin{eqnarray*}
\forall x &:& ((\text{Continent}(x) \myand \mynot (\text{Australia}(x)) \myand \mynot (\text{Antarctica}(x))) \implies \\
&& ~~\exists y : (\text{Continent}(y) \myand \text{Connected}(x, y)))
\end{eqnarray*}


\paragraph{b)} Every person who is smart and studies hard will get a higher score than every person who is not smart and does not study hard.

\begin{eqnarray*}
\forall x, y &:& (\text{Person}(x) \myand \text{Person}(y) \implies \\
&& ~~(\text{Smart}(x) \myand \text{StudiesHard}(x) \myand \\
&& ~~\mynot(\text{Smart}(y)) \myand \mynot(\text{StudiesHard}(y)) \implies \\
&& ~~~~\text{GetsHigherScoreThan}(x,y)))
\end{eqnarray*}

\paragraph{c)} Everything that walks like a duck and talks like a duck is either a duck or a human imitating a duck.

\begin{eqnarray*}
\forall x &:& ((\text{WalksLikeDuck}(x) \myand \text{TalksLikeDuck}(x)) \implies \\
&& ~~ (\text{Duck}(x) \myor (\text{Human}(x) \myand \text{ImitatingDuck}(x))))
\end{eqnarray*}

\end{answer}

\paragraph{d)} A gold medal is worth more than a silver medal if they are medals in the same event.

\begin{eqnarray*}
\forall x, y &:& ((\text{Medal}(x) \myand \text{Medal}(y) \myand \mynot(x=y)) \implies \\
&& ~~(\text{Gold}(x) \myand \text{Silver}(y) \myand (\text{Event}(x) = \text{Event}(y)) \implies \\
&& ~~~~\text{WorthMoreThan}(x, y))
)
\end{eqnarray*}

\paragraph{e)} Every thing that loves all humans is a dog.

\begin{eqnarray*}
\forall x, y &:& (\text{Human}(x) \myand \text{Loves}(y, x) \implies (\text{Dog}(x)))
\end{eqnarray*}

\paragraph{f)} There is a dog that does not love all humans.

\begin{eqnarray*}
\exists x &:& \exists y : (\text{Dog}(x) \myand \text{Human}(y) \myand \mynot(\text{Loves}(x, y)))
\end{eqnarray*}

\paragraph{g)} Every thing that is an enemy of some thing that is an enemy of me is a friend of me.

\begin{eqnarray*}
\forall x, y &:& (\text{EnemyOf}(x, \text{Me}) \myand \text{EnemyOf}(y, x) \implies \text{FriendOf}(y, \text{Me}))
\end{eqnarray*}

\begin{answer}{2}

Given: 
\begin{eqnarray*}
\forall x, y &:& \text{LovesTheCombinationOf}(\text{John}, x, y) \myor \text{MakesSick}(x, \text{John}) \myor \text{RuinsTasteOf}(y, x) \\
\forall v, w &:& \mynot \text{LovesTheCombinationOf}(v, \text{Rice}, w) \myor \text{Flavorful}(w)
\end{eqnarray*}

Substitution: $\{v/\text{John}, w/z, x/\text{Rice}, y/z \}$.

\begin{eqnarray*}
\forall z &:& (\text{LovesTheCombinationOf}(\text{John}, \text{Rice}, z) \myor \text{MakesSick}(\text{Rice}, \text{John}) \myor \text{RuinsTasteOf}(z, \text{Rice})) \myand \\
&& (\mynot \text{LovesTheCombinationOf}(\text{John}, \text{Rice}, z) \myor \text{Flavorful}(z))
\end{eqnarray*}

% note: can we drop things that don't depend on z?
Resolution:

\begin{eqnarray*}
\forall z &:& \text{MakesSick}(\text{Rice}, \text{John}) \myor \text{RuinsTasteOf}(z, \text{Rice})) \myor \text{Flavorful}(z)
\end{eqnarray*}

\end{answer}

\begin{answer}{3}

Adding the negated sentence to the knowledge base and substituting Skolem functions for universals yields the following.

\begin{eqnarray}
&& (\mynot \text{EnemyOf}(y, x) \myor \mynot \text{EnemyOf}(z, y) \myor \text{FriendOf}(z, x)) \myand \\
&& (\mynot \text{EnemyOf}(y, x) \myor \text{EnemyOf}(x, y)) \myand \\
&& \text{EnemyOf}(a(x), x) \myand \\
&& \text{EnemyOf}(b(x), y) \myand \\
&& (\mynot (b(x) = a(x) )) \myand \\
&& \text{FriendOf}(c(x), x) \myand \\
&& (x = c(x)) \myand \\
&& (\mynot \text{EnemyOf}(z, a(x)) \myor \text{FriendOf}(z,x))~~(\text{by 1 and 3 with subst} \{ y/a(x) \})
\end{eqnarray}

\end{answer}
