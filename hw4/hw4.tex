\begin{answer}{1}

We can find $P(+u | +e )$ by finding $P(+u , +e )$ and $P(-u , +e )$.

\begin{eqnarray*}
P(+u, +e) &=& \sum_t P(+e | +u, t) \sum_i P(t | i) P(i) \sum_h P(+u | i, h) P(h) \\
          &=& \sum_t P(+e | +u, t) \sum_i P(t | i) P(i) f_1(i) \\
          &=& \sum_t P(+e | +u, t) f_2(t) \\
          &=& f_3 \\
f_1(-i) &=& P(+u | -i, -h)P(-h) + P(+u|-i, +h)P(+h) \\
        &=& 0.1 (0.4) + 0.5 (0.6) = 0.34 \\
f_1(+i) &=& P(+u | +i, -h)P(-h) + P(+u|+i, +h)P(+h) \\
        &=& 0.3 (0.4) + 0.9 (0.6) = 0.66 \\
f_2(+t) &=& P(+t | -i)f_1(-i)P(-i) + P(+t | +i)f_1(+i)P(+i) \\
        &=& .119 + .1584 = 0.2774 \\
f_2(-t) &=& P(-t | -i)f_1(-i)P(-i) + P(-t | +i)f_1(+i)P(+i) \\
        &=& .119 + .0396 = 0.1586 \\ 
f_3 &=& P(+e | +u, +t)f_2(+t) + P(+e | +u, -t)f_2(-t) \\
    &=& 0.9 (0.2774) + 0.7 (0.11102) \\
    &=& 0.36068  \\
P(+u, +e) &=& 0.36068 
\end{eqnarray*}

\begin{eqnarray*}
P(-u, +e) &=& \sum_t P(+e | -u, t) \sum_i P(t | i) P(i) \sum_h P(-u | i, h) P(h) \\
          &=& \sum_t P(+e | -u, t) \sum_i P(t | i) P(i) f_1(i) \\
          &=& \sum_t P(+e | -u, t) f_2(t) \\
          &=& f_3 \\
f_1(-i) &=& P(-u | -i, -h)P(-h) + P(-u|-i, +h)P(+h) \\
        &=& 0.9 (0.4) + 0.5 (0.6) = 0.66 \\
f_1(+i) &=& P(-u | +i, -h)P(-h) + P(-u|+i, +h)P(+h) \\
        &=& 0.7 (0.4) + 0.1 (0.6) = 0.34 \\
f_2(+t) &=& P(+t | -i)f_1(-i)P(-i) + P(+t | +i)f_1(+i)P(+i) \\
        &=& .099 + .1904 = .2894 \\
f_2(-t) &=& P(-t | -i)f_1(-i)P(-i) + P(-t | +i)f_1(+i)P(+i) \\
        &=& .099 + .0476 = 0.1466 \\ 
f_3 &=& P(+e | -u, +t)f_2(+t) + P(+e | -u, -t)f_2(-t) \\
    &=& 0.5 (0.2894) + 0.3 (0.1466) \\
    &=& 0.18868  \\
P(-u, +e) &=& 0.18868
\end{eqnarray*}


\begin{eqnarray*}
P(+u | +e ) &=& {P(+u, +e) \over P(+e)} \\
&=& {P(+u, +e) \over P(+u, +e)+ P(-u, +e)} \\
&=& 0.36068 \over 0.36068 + 0.18868 \\
&=& 0.36068 \over 0.54936 \\
&=& 0.6565458 \approx 0.66
\end{eqnarray*}


\end{answer}
